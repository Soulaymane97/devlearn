/* CSS moderne avec variables */ :root { --primary: #7c3aed; --bg: #0a0a0f; --text: #e8e8f0; } .container { display: grid; grid-template-columns: repeat(3, 1fr); gap: 1.5rem; padding: 2rem; } .card { background: var(--bg); border-radius: 12px; transition: transform 0.3s; } .card:hover { transform: translateY(-4px); box-shadow: 0 20px 40px rgba(0,0,0,0.3); }
// JavaScript moderne — Fetch API const fetchData = async () => { try { const response = await fetch('/api/articles'); const data = await response.json(); data.forEach(article => { const card = document.createElement('div'); card.className = 'card'; card.innerHTML = ` <h3>${article.title}</h3> <p>${article.excerpt}</p> `; document.querySelector('.grid') .appendChild(card); }); } catch (err) { console.error('Erreur:', err); } }; fetchData();
# Python 3 — Classe et méthodes class Article: def __init__(self, titre, contenu): self.titre = titre self.contenu = contenu self.vues = 0 def afficher(self): return f"📄 {self.titre} ({self.vues} vues)" def incrementer_vues(self): self.vues += 1 # Utilisation article = Article( titre="Apprendre Python", contenu="Python est simple..." ) article.incrementer_vues() print(article.afficher()) # → 📄 Apprendre Python (1 vues)
# Django — Modèle, Vue, URL # models.py from django.db import models class Article(models.Model): titre = models.CharField(max_length=200) contenu = models.TextField() date = models.DateTimeField(auto_now_add=True) def __str__(self): return self.titre # views.py from django.shortcuts import render from .models import Article def liste_articles(request): articles = Article.objects.all() return render(request, 'articles/liste.html', {'articles': articles} ) # urls.py path('articles/', views.liste_articles)
